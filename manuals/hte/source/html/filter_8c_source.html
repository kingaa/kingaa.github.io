55 static inline double prediction
56 (double t1, double t2, double p, double lambda, double gamma)
57 {
58  if (t2 > t1) {
59  double s = lambda+gamma;
60  if (s > 0) {
61  double a = exp(-s*(t2-t1));
62  p = (1-a)*lambda/s + a*p;
63  }
64  }

80 static inline double filtration
81 (int result, double p, double Alpha, double Beta, double Eta, double *ll)
82 {
83  if (result != NA_INTEGER) {
84  if (result == 2) { // isolation effect
85  p = (Eta*p)/(1-p+Eta*p);
86  } else {
87  double prob[2], lik;
88  test_prob(result, Alpha, Beta, prob);
89  prob[0] *= 1 - p;
90  prob[1] *= p;
91  lik = prob[0] + prob[1];
92  p = prob[1] / lik;
93  *ll += log(lik);
94  }
95  }
96  return p;
97 }

155  int i = locat + event*dim[1]; // current unit & event

157  // assimilate new test result:
