268  // drop the node holding black ball a.
269  void drop (ball_t *a) {
270  if (!a->is(black))
271  err("in '%s': inconceivable! (color: %s)",__func__,colores[a->color]); // #nocov
272  node_t *p = a->holder();
273  if (p->size() > 2) { // pocket is large: we simply drop the ball
274  p->erase(a);
275  delete a;
276  } else { // pocket contains two: action depends on the other ball
277  ball_t *b = p->other(a);
278  switch (b->color) {
279  case blue: // change black ball for red ball
280  a->color = red;
281  break;
282  case purple: // swap black ball for green ball, delete node
283  a->deme() = p->deme;
284  swap(a,p->green_ball());
285  destroy_node(p);
286  drop(a); // recursively pursue dropping ball a
287  break;
288  case black: case green: // swap other for green, delete node
289  swap(b,p->green_ball());
290  destroy_node(p);
291  break;
292  case red: case grey: // #nocov
293  err("in '%s': inconceivable error.",__func__); // #nocov
294  break;
295  }
296  }
