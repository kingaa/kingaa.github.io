276  void drop (ball_t *a) {
277  if (!a->is(black))
278  err("in '%s': inconceivable! (color: %s)",__func__,colores[a->color]); // #nocov
279  node_t *p = a->holder();
280  if (p->size() > 2) { // pocket is large: we simply drop the ball
281  p->erase(a);
282  delete a;
283  } else { // pocket contains two: action depends on the other ball
284  ball_t *b = p->other(a);
285  switch (b->color) {
286  case blue: // change black ball for red ball
287  a->color = red;
288  break;
289  case purple: // swap black ball for green ball, delete node
290  a->deme() = p->deme;
291  swap(a,p->green_ball());
292  destroy_node(p);
293  drop(a); // recursively pursue dropping ball a
294  break;
295  case black:
296  if (b->deme() == p->deme) { // swap other for green, delete node
297  swap(b,p->green_ball());
298  destroy_node(p);
299  } else { // deme changes here, insert purple ball
300  a->color = purple;
301  }
302  break;
303  case green:
304  if (b->owner()->deme == p->deme) { // swap other for green, delete node
305  swap(b,p->green_ball());
306  destroy_node(p);
307  } else {
308  a->color=purple;
309  }
310  break;
311  case red: case grey: // #nocov
312  err("in '%s': inconceivable error.",__func__); // #nocov
313  break;
314  }
