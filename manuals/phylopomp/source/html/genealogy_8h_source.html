175 void lineage_count (double *tout, int *deme,
176 int *ell, int *sat, int *etype) const {
177 slate_t tcur = timezero();
178 for (size_t j = 0; j < _ndeme; j++) {
179 tout[j] = tcur;
180 deme[j] = j+1;
181 sat[j] = ell[j] = 0;
182 etype[j] = 0;
183 }
184 for (const node_t *p : *this) {
185 if (tcur < p->slate) {
186 tout += _ndeme; ell += _ndeme; sat += _ndeme;
187 deme += _ndeme; etype += _ndeme;
188 tcur = p->slate;
189 for (size_t j = 0; j < _ndeme; j++) {
190 tout[j] = tcur;
191 deme[j] = j+1;
192 ell[j] = (ell-_ndeme)[j];
193 sat[j] = 0;
194 etype[j] = 0;
195 }
196 }
197 p->lineage_incr(ell,sat,etype);
198 }
199 tout += _ndeme; ell += _ndeme; sat += _ndeme;
200 deme += _ndeme; etype += _ndeme;
201 tcur = time();
202 for (size_t j = 0; j < _ndeme; j++) {
203 tout[j] = tcur;
204 sat[j] = ell[j] = 0;
205 deme[j] = j+1;
206 etype[j] = 3;
207 }

234 void gendat (double *tout, int *anc, int *lin,
235 int *sat, int *type,
236 int *index, int *child) const {
237 int m, n, k;
238 node_it i, j;
239 for (k = 0, n = 0, i = begin(); i != end(); i++, n++) {
240 node_t *p = *i;
241 assert(!p->holds(black)); // tree should be pruned first
242 tout[n] = p->slate;
243 if (p->is_root()) {
244 type[n] = 0; // root node
245 } else if (p->holds(blue)) {
246 type[n] = 1; // sample node
247 } else {
248 type[n] = 2; // internal node
249 }
250 lin[n] = p->lineage(); // 0-based indexing
251 sat[n] = p->nchildren();
252 index[n] = k;
253 k += sat[n];
254 child[n] = NA_INTEGER;
255 if (p->is_root()) {
256 anc[n] = n; // 0-based indexing
257 } else {
258 for (m = 0, j = begin(); j != i; j++, m++) {
259 node_t *q = *j;
260 if (p->parent()->uniq == q->uniq) {
261 anc[n] = m;
262 break;
263 }
264 }
265 }
266 }
267 tout[n] = time();
268 for (k = 0, n = 0, i = begin(); i != end(); i++, n++) {
269 node_t *p = *i;
270 j = i; j++;
271 for (m = n+1; j != end(); m++, j++) {
272 node_t *q = *j;
273 if (p->uniq == q->parent()->uniq) {
274 child[k++] = m;
275 }
276 }
277 }
278 };

508 void curtail (slate_t tnew, slate_t troot) {
509 if (tnew < troot) troot = tnew;
510 if (!empty() && tnew < time()) {
511 node_t *p = back();
512 while (!empty() && p->slate > tnew) {
513 ball_t *b;
514 while (p->size() > 1) {
515 b = p->last_ball();
516 switch (b->color) {
517 case black:
518 p->erase(b); delete b;
519 break;
520 case green: case blue: // #nocov
521 assert(0); // #nocov
522 break; // #nocov
523 }
524 }
525 b = p->last_ball();
526 switch (b->color) {
527 case blue:
528 b->color = black;
529 case black:
530 b->deme() = p->deme();
531 swap(b,p->green_ball());
532 case green:
533 destroy_node(p);
534 break;
535 }
536 if (!empty()) p = back();
537 }
538 }
539 time() = tnew;
540 if (!empty() && troot > timezero()) {
541 node_t *p = front();
542 node_t *q;
543 while (!empty() && p->slate <= troot) {
544 ball_t *b;
545 assert(p->holds_own());
546 while (p->size() > 1) {
547 b = p->last_ball();
548 switch (b->color) {
549 case blue:
550 p->erase(b); delete b;
551 break;
552 case black:
553 q = make_node(b->deme());
554 q->slate = troot;
555 b->holder() = q;
556 q->insert(b); p->erase(b);
557 push_back(q);
558 break;
559 case green:
560 q = b->child();
561 if (q->slate < troot) {
562 q->insert(b); p->erase(b);
563 b->holder() = q;
564 } else {
565 node_t *pp = make_node(b->deme());
566 pp->slate = troot;
567 pp->insert(b); p->erase(b);
568 b->holder() = pp;
569 push_back(pp);
570 }
571 break;
572 }
573 }
574 destroy_node(p);
575 if (!empty()) p = front();
576 }
577 sort();
578 }
579 if (troot > timezero()) timezero() = troot;
580 };

602 size_t scan_label (const std::string& s, color_t* col,
603 name_t *deme, slate_t *time) const {
604 size_t n = s.size();
605 if (n < 1)
606 err("in '%s' (%s line %d): invalid Newick format: empty label.",__func__,__FILE__,__LINE__);
607 size_t sz, i = 1;
608 switch (s[0]) {
609 case 'o':
610 *col = black;
611 break;
612 case 'b':
613 *col = blue;
614 break;
615 case 'g': case 'm':
616 *col = green;
617 break;
618 default:
619 err("in '%s' (%s line %d): invalid Newick label: expected one of 'b','g','m', or 'o', got '%c'.",
620 __func__,__FILE__,__LINE__,s[0]);
621 break;
622 }
623 while (i < n && s[i] == '_') i++;
624 if (i == n)
625 err("in '%s': invalid Newick format: premature termination.",__func__);
626 if (s[i] == '(' || s[i] == ')' || s[i] == ',' || s[i] == ';')
627 err("in '%s' (%s line %d): invalid Newick format.",__func__,__FILE__,__LINE__);
628 if (s[i] == ':') {
629 *deme = 0;
630 } else {
631 try {
632 *deme = name_t(stoi(s.substr(i),&sz));
633 i += sz;
634 }
635 catch (const std::invalid_argument& e) {
636 err("in '%s' (%s line %d): invalid Newick format: deme should be indicated with an integer.",
637 __func__,__FILE__,__LINE__);
638 }
639 catch (const std::out_of_range& e) {
640 err("in '%s': invalid Newick format: deme out of range.",__func__);
641 }
642 catch (const std::exception& e) {
643 err("in '%s': parsing deme label: %s.",__func__,e.what());
644 }
645 catch (...) {
646 err("in '%s': other deme-parsing error.",__func__);
647 }
648 }
649 // skip to branch length
650 while (i < n && s[i] != ':' &&
651 s[i] != '(' && s[i] != ')' && s[i] != ',' && s[i] != ';') i++;
652 if (i == n || s[i] != ':')
653 err("in '%s': invalid Newick format: missing or invalid branch length.",__func__);
654 i++;
655 try {
656 *time = slate_t(stod(s.substr(i),&sz));
657 }
658 catch (const std::invalid_argument& e) {
659 err("in '%s': invalid Newick format: branch length should be a non-negative decimal number.",__func__);
660 }
661 catch (const std::out_of_range& e) {
662 err("in '%s': invalid Newick format: branch length out of range.",__func__);
663 }
664 catch (const std::exception& e) {
665 err("in '%s': parsing branch-length: %s.",__func__,e.what());
666 }
667 catch (...) {
668 err("in '%s': other branch-length parsing error.",__func__);
669 }
670 i += sz;
671 return i;
672 };

715 size_t scan_tree (const std::string& s,
716 const slate_t t0, node_t **root) {
717 size_t n = s.size();
718 size_t i = 1, j = 1, k;
719 size_t stack = 1;
720 while (j < n && stack > 0) {
721 switch (s[j]) {
722 case '(':
723 stack++;
724 break;
725 case ')':
726 stack--;
727 break;
728 default:
729 break;
730 }
731 j++;
732 }
733 if (stack > 0)
734 err("in '%s': premature end of Newick string.",__func__);
735 node_t *p = 0;
736 k = j;
737 k += scan_node(s.substr(j),t0,&p);
738 parse(s.substr(i,j-i-1),p->slate,p);
739 *root = p;
740 return k;
741 };

746 genealogy_t& parse (const std::string& s, slate_t t0, node_t *p = 0) {
747 size_t i = 0;
748 size_t n = s.size();
749 while (i < n) {
750 switch (s[i]) {
751 case '(':
752 {
753 genealogy_t G(t0,unique());
754 node_t *q = 0;
755 i += G.scan_tree(s.substr(i),t0,&q);
756 *this += G;
757 if (p != 0) {
758 ball_t *g = q->green_ball();
759 q->erase(g); p->insert(g); g->holder() = p;
760 }
761 }
762 break;
763 case 'b':
