87// This function draws a random increment of a gamma whitenoise process.
88// This will have expectation=dt and variance=(sigma^2*dt)
89// If dW = rgammawn(sigma,dt), then
90// mu dW/dt is a candidate for a random rate process within an
91// Euler-multinomial context, i.e.,
92// E[mu*dW] = mu*dt and Var[mu*dW] = mu*sigma^2*dt
93SEXP R_GammaWN (SEXP n, SEXP sigma, SEXP deltat) {
94 int k, nval, nsig, ndt;
95 double *x, *sig, *dt;
96 SEXP ans;
97 PROTECT(n = AS_INTEGER(n));
98 nval = INTEGER(n)[0];
99 PROTECT(sigma = AS_NUMERIC(sigma));
100 nsig = LENGTH(sigma);
101 sig = REAL(sigma);
102 PROTECT(deltat = AS_NUMERIC(deltat));
103 ndt = LENGTH(deltat);
104 dt = REAL(deltat);
105 PROTECT(ans = NEW_NUMERIC(nval));
106 x = REAL(ans);
107 GetRNGstate();
108 for (k = 0; k < nval; k++) {
109 x[k] = rgammawn(sig[k%nsig],dt[k%ndt]);
110 }
111 PutRNGstate();
112 UNPROTECT(4);
113 return ans;
114}
