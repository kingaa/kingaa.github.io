57 static R_INLINE SEXP matchnames
58 (
59  SEXP provided, SEXP needed, const char *where
60  ) {
61  int m = LENGTH(provided);
62  int n = length(needed);
63  SEXP index;
64  int *idx, i, j;
65  PROTECT(provided = AS_CHARACTER(provided));
66  PROTECT(needed = AS_CHARACTER(needed));
67  if (invalid_names(provided))
68  err("invalid variable names among the %s.",where); // #nocov
69  PROTECT(index = NEW_INTEGER(n));
70  idx = INTEGER(index);
71  for (i = 0; i < n; i++) {
72  for (j = 0; j < m; j++) {
73  if (!strcmp(CHAR(STRING_ELT(provided,j)),CHAR(STRING_ELT(needed,i)))) {
74  idx[i] = j;
75  break;
76  }
77  }
78  if (j==m)
79  err("variable '%s' not found among the %s.",CHAR(STRING_ELT(needed,i)),where);
80  }
81  UNPROTECT(3);
82  return index;
83 }
